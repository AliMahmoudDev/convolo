    58→/**
    59→ * Build the system prompt for a conversation.
    60→ * (Backwards compatible with existing code)
    61→ */
    62→export function buildSystemPrompt(
    63→  targetLanguage: string,
    64→  nativeLanguage: string,
    65→  proficiencyLevel: ProficiencyLevel,
    66→  scenarioContext?: string
    67→): string {
    68→  const targetName = LANGUAGE_NAMES[targetLanguage] || targetLanguage;
    69→  const nativeName = LANGUAGE_NAMES[nativeLanguage] || nativeLanguage;
    70→  const difficultyInstructions = DIFFICULTY_INSTRUCTIONS[proficiencyLevel];
    71→
    72→  return `You are Convolo, an expert ${targetName} language tutor. You are having a conversation with a learner whose native language is ${nativeName}.
    73→
    74→## YOUR ROLE
    75→- You are a friendly, encouraging conversation partner who helps learners practice ${targetName}.
    76→- You ONLY speak in ${targetName} (except for explanations and translations, which are in ${nativeName}).
    77→- You stay in character at all times — you are a language tutor, not a general AI assistant.
    78→- You keep the conversation going naturally — ask follow-up questions, show interest, and guide the dialogue.
    79→
    80→## DIFFICULTY: ${proficiencyLevel.toUpperCase()}
    81→${difficultyInstructions}
    82→
    83→${scenarioContext ? `## SCENARIO\n${scenarioContext}\n- Stay in the context of this scenario.\n- If the user goes off-topic, gently guide them back.\n\n` : ""}## RESPONSE FORMAT (MANDATORY — ALWAYS respond in this exact JSON format)
    84→You MUST respond with a valid JSON object. No markdown, no code fences, just pure JSON.
    85→
    86→{
    87→  "reply": "Your conversational response in ${targetName}",
    88→  "translatedReply": "Translation of your reply in ${nativeName}",
    89→  "corrections": [
    90→    {
    91→      "original": "the exact text the user wrote that was WRONG",
    92→      "corrected": "the correct version",
    93→      "explanation": "Why it was wrong, explained in ${nativeName}",
    94→      "severity": "minor|moderate|major"
    95→    }
    96→  ],
    97→  "hints": [
    98→    {
    99→      "original": "what the user wrote (not wrong, just could be better)",
   100→      "suggested": "a more natural/correct version",
   101→      "explanation": "Why this is better, explained in ${nativeName}"
   102→    }
   103→  ],
   104→  "vocabularyItems": [
   105→    {
   106→      "word": "a useful word from this exchange",
   107→      "translation": "translation in ${nativeName}",
   108→      "definition": "definition in ${targetName}",
   109→      "partOfSpeech": "noun|verb|adjective|adverb|etc",
   110→      "exampleSentence": "an example sentence using this word"
   111→    }
   112→  ],
   113→  "grammarNotes": [
   114→    {
   115→      "rule": "the grammar rule name",
   116→      "explanation": "explanation in ${nativeName}",
   117→      "example": "correct usage example"
   118→    }
   119→  ],
   120→  "suggestions": ["suggested reply 1 in ${targetName}", "suggested reply 2 in ${targetName}"]
   121→}
   122→
   123→## RULES
   124→1. "reply" is ALWAYS in ${targetName}. Never in ${nativeName}.
   125→2. "translatedReply" is ALWAYS in ${nativeName}. This helps the learner understand your response.
   126→3. "corrections" — ONLY for REAL ERRORS (wrong grammar, wrong word, wrong conjugation, etc.). If the user's message is PERFECT with NO mistakes, return an EMPTY array []. Do NOT add a correction like "No correction needed" or "No errors" — just return []. Only include REAL corrections where the original text was actually wrong and needed fixing.
   127→4. "hints" — Style suggestions that are NOT errors and truly help the learner. For example: more natural phrasing alternatives ("I am happy" → "I'm happy" for casual speech), better word choices ("very big" → "huge"), or register improvements. Do NOT flag apostrophe/punctuation differences — these are typing style, not language learning issues. Empty array [] if none.
   128→5. "vocabularyItems" — include 1-3 useful words from this exchange. Maximum 3. Empty array [] if none.
   129→6. "grammarNotes" — include only if a grammar rule is relevant. Maximum 1. Empty array [] if none.
   130→7. "suggestions" — provide 2-3 short suggested replies in ${targetName} the student could use next. This helps beginners who don't know what to say.
   131→8. Severity: "minor" = typo/small error, "moderate" = grammar error that changes meaning slightly, "major" = error that significantly changes meaning or is a fundamental mistake.
   132→9. IMPORTANT: Know the difference between corrections, hints, and things to ignore:
   133→   - "I are happy" → "I am happy" is a CORRECTION (wrong grammar)
   134→   - "She dont know" → "She doesn't know" is a CORRECTION (wrong grammar)
   135→   - "Iam" or "Im" → "I'm" — IGNORE completely! This is just a typing style difference (apostrophe missing). Do NOT put it in corrections OR hints. The meaning is perfectly clear.
   136→   - "I am happy" → "I'm happy" can be a HINT (informal/casual form is more natural in conversation)
   137→   - "very big" → "huge" can be a HINT (better vocabulary choice)
   138→   - NEVER flag missing apostrophes in contractions as either corrections or hints. They are typing preferences, not language errors.
   139→10. CRITICAL: ONLY correct the user's CURRENT/LATEST message. NEVER re-correct mistakes from previous messages that were already corrected. Each response should only check the most recent user message.
   140→11. NEVER include markdown formatting, code fences, or any text outside the JSON object.
   141→12. If the user says something inappropriate, respond politely in ${targetName} and redirect to the lesson. Set corrections to [] and include no vocabulary or grammar.
   142→13. Keep the conversation engaging — don't just correct, also respond naturally to what they said.
   143→14. CRITICAL: The user is LEARNING. Even small mistakes like wrong gender agreement, missing accents, wrong verb conjugation, wrong word order — ALL of these should be corrected, especially for beginners.
   144→
   145→## SAFETY
   146→- Never discuss harmful, illegal, or explicit topics.
   147→- Never break character as a language tutor.
   148→- If the user asks non-language questions, gently redirect: "Let's keep practicing ${targetName}!"`;
   149→}

   176→export function parseAIResponse(rawText: string): ParsedAIResponse {
   177→  const emptyResponse: ParsedAIResponse = {
   178→    reply: "",
   179→    corrections: [],
   180→    hints: [],
   181→    vocabularyItems: [],
   182→    grammarNotes: [],
   183→    suggestions: [],
   184→    translatedReply: "",
   185→  };
   186→
   187→  if (!rawText || !rawText.trim()) {
   188→    return emptyResponse;
   189→  }
   190→
   191→  let jsonStr = rawText.trim();
   192→
   193→  // Strip markdown code fences if present
   194→  const codeFenceMatch = jsonStr.match(/```(?:json)?\s*\n?([\s\S]*?)\n?\s*```/);
   195→  if (codeFenceMatch) {
   196→    jsonStr = codeFenceMatch[1].trim();
   197→  }
   198→
   199→  // Try to find JSON object in the text
   200→  const jsonStart = jsonStr.indexOf("{");
   201→  const jsonEnd = jsonStr.lastIndexOf("}");
   202→
   203→  if (jsonStart === -1 || jsonEnd === -1 || jsonEnd < jsonStart) {
   204→    console.warn("[AI Parser] No JSON object found in AI response. Using raw text as reply.");
   205→    return {
   206→      ...emptyResponse,
   207→      reply: rawText.trim(),
   208→    };
   209→  }
   210→
   211→  jsonStr = jsonStr.slice(jsonStart, jsonEnd + 1);
   212→
   213→  try {
   214→    const parsed = JSON.parse(jsonStr);
   215→
   216→    // Parse corrections and hints from AI response
   217→    const rawCorrections = parseCorrections(parsed.corrections);
   218→    const rawHints = parseHints(parsed.hints);
   219→
   220→    // SMART FILTER: Remove corrections/hints that are just typing style
   221→    // (apostrophe differences, capitalization) — not worth showing
   222→    const realCorrections = rawCorrections.filter(
   223→      (c) => !shouldIgnoreCorrection(c.original, c.corrected)
   224→    );
   225→    const realHints = rawHints.filter(
   226→      (h) => !shouldIgnoreCorrection(h.original, h.suggested)
   227→    );
   228→
   229→    return {
   230→      reply: typeof parsed.reply === "string" ? parsed.reply : "",
   231→      translatedReply: typeof parsed.translatedReply === "string" ? parsed.translatedReply : "",
   232→      corrections: realCorrections,
   233→      hints: realHints,
   234→      vocabularyItems: parseVocabularyItems(parsed.vocabularyItems),
   235→      grammarNotes: parseGrammarNotes(parsed.grammarNotes),
   236→      suggestions: parseSuggestions(parsed.suggestions),
   237→    };
   238→  } catch (error) {
   239→    console.error("[AI Parser] Failed to parse AI JSON response:", error);
   240→    return {
   241→      ...emptyResponse,
   242→      reply: rawText.trim(),
   243→    };
   244→  }
   245→}
   246→
   247→function parseCorrections(raw: unknown): Correction[] {
   248→  if (!Array.isArray(raw)) return [];
   249→  return raw
   250→    .filter((c): c is Record<string, unknown> => typeof c === "object" && c !== null)
   251→    .map((c) => ({
   252→      original: String(c.original || ""),
   253→      corrected: String(c.corrected || ""),
   254→      explanation: String(c.explanation || ""),
   255→      severity: ["minor", "moderate", "major"].includes(c.severity as string)
   256→        ? (c.severity as "minor" | "moderate" | "major")
   257→        : "moderate",
   258→    }))
   259→    .filter((c) => {
   260→      // Must have both original and corrected text
   261→      if (!c.original || !c.corrected) return false;
   262→      // Filter out fake corrections where original === corrected (no actual change)
   263→      if (c.original.trim().toLowerCase() === c.corrected.trim().toLowerCase()) return false;
   264→      // Filter out placeholder corrections like "No correction needed", "No errors", etc.
   265→      const noCorrectionPhrases = [
   266→        "no correction needed",
   267→        "no corrections needed",
   268→        "no errors",
   269→        "no error",
   270→        "no mistakes",
   271→        "correct",
   272→        "perfect",
   273→        "no change needed",
   274→        "n/a",
   275→        "none",
   276→      ];
   277→      const originalLower = c.original.trim().toLowerCase();
   278→      const correctedLower = c.corrected.trim().toLowerCase();
   279→      if (noCorrectionPhrases.some((phrase) => originalLower === phrase || correctedLower === phrase)) return false;
   280→      return true;
   281→    });
   282→}
   283→
   284→/**
   285→ * Check if a correction should be COMPLETELY IGNORED.
   286→ * Apostrophe/contraction differences are just typing style —
   287→ * not errors, not even hints. The meaning is perfectly clear.
   288→ * Also ignores capitalization-only differences.
   289→ */
   290→function shouldIgnoreCorrection(original: string, corrected: string): boolean {
   291→  const orig = original.trim();
   292→  const corr = corrected.trim();
   293→
   294→  // Check if only capitalization changed
   295→  if (orig.toLowerCase() === corr.toLowerCase()) {
   296→    return true;
   297→  }
   298→
   299→  // Remove all apostrophes, hyphens, and lowercase everything
   300→  // If they're the same after that, it's just an apostrophe/punctuation difference
   301→  const normalize = (s: string) =>
   302→    s.toLowerCase().replace(/[''`ʼ‛-]/g, "").replace(/\s+/g, " ").trim();
   303→
   304→  if (normalize(orig) === normalize(corr)) {
   305→    return true; // Only difference is apostrophe/punctuation — IGNORE
   306→  }
   307→
   308→  // Split into words and check word-by-word
   309→  // Handles multi-word cases like "Iam here" → "I'm here"
   310→  const origWords = orig.split(/\s+/);
   311→  const corrWords = corr.split(/\s+/);
   312→
   313→  if (origWords.length === corrWords.length) {
   314→    let allWordsAreTypingOnly = true;
   315→    for (let i = 0; i < origWords.length; i++) {
   316→      const ow = origWords[i].toLowerCase().replace(/[''`ʼ‛-]/g, "");
   317→      const cw = corrWords[i].toLowerCase().replace(/[''`ʼ‛-]/g, "");
   318→
   319→      if (ow === cw) {
   320→        continue; // Same word after removing apostrophes
   321→      }
   322→
   323→      // Contraction without apostrophe: "iam" normalizes to "im"
   324→      const contractionMap: Record<string, string> = {
   325→        "iam": "im", "im": "im", "ive": "ive", "id": "id", "ill": "ill",
   326→        "youre": "youre", "youve": "youve", "youll": "youll", "youd": "youd",
   327→        "hes": "hes", "shes": "shes", "its": "its", "were": "were",
   328→        "theyre": "theyre", "theyve": "theyve", "theyll": "theyll", "theyd": "theyd",
   329→        "dont": "dont", "doesnt": "doesnt", "didnt": "didnt", "wont": "wont",
   330→        "wouldnt": "wouldnt", "couldnt": "couldnt", "shouldnt": "shouldnt",
   331→        "isnt": "isnt", "arent": "arent", "wasnt": "wasnt", "werent": "werent",
   332→        "havent": "havent", "hasnt": "hasnt", "hadnt": "hadnt", "cant": "cant",
   333→        "mustnt": "mustnt", "lets": "lets", "whos": "whos", "whats": "whats",
   334→      };
   335→
   336→      const owNorm = contractionMap[ow] || ow;
   337→      const cwNorm = contractionMap[cw] || cw;
   338→
   339→      if (owNorm !== cwNorm) {
   340→        allWordsAreTypingOnly = false;
   341→        break;
   342→      }
   343→    }
   344→
   345→    if (allWordsAreTypingOnly) {
   346→      return true; // All differences are just apostrophe/typing style — IGNORE
   347→    }
   348→  }
   349→
   350→  return false;
   351→}

   353→function parseHints(raw: unknown): Hint[] {
   354→  if (!Array.isArray(raw)) return [];
   355→  return raw
   356→    .filter((h): h is Record<string, unknown> => typeof h === "object" && h !== null)
   357→    .map((h) => ({
   358→      original: String(h.original || ""),
   359→      suggested: String(h.suggested || h.corrected || ""),
   360→      explanation: String(h.explanation || ""),
   361→    }))
   362→    .filter((h) => h.original && h.suggested);
   363→}
   364→
   365→function parseVocabularyItems(raw: unknown): VocabularyExtraction[] {
   366→  if (!Array.isArray(raw)) return [];
   367→  return raw
   368→    .filter((v): v is Record<string, unknown> => typeof v === "object" && v !== null)
   369→    .map((v) => ({
   370→      word: String(v.word || ""),
   371→      translation: String(v.translation || ""),
   372→      definition: String(v.definition || ""),
   373→      partOfSpeech: String(v.partOfSpeech || ""),
   374→      exampleSentence: String(v.exampleSentence || ""),
   375→    }))
   376→    .filter((v) => v.word && v.translation);
   377→}
   378→
   379→function parseGrammarNotes(raw: unknown): GrammarNote[] {
   380→  if (!Array.isArray(raw)) return [];
   381→  return raw
   382→    .filter((g): g is Record<string, unknown> => typeof g === "object" && g !== null)
   383→    .map((g) => ({
   384→      rule: String(g.rule || ""),
   385→      explanation: String(g.explanation || ""),
   386→      example: String(g.example || ""),
   387→    }))
   388→    .filter((g) => g.rule && g.explanation);
   389→}
   390→
   391→function parseSuggestions(raw: unknown): string[] {
   392→  if (!Array.isArray(raw)) return [];
   393→  return raw
   394→    .filter((s): s is string => typeof s === "string" && s.trim().length > 0)
   395→    .map((s) => s.trim())
   396→    .slice(0, 5); // Max 5 suggestions
   397→}

   481→// OpenAI-compatible provider call (Groq, xAI, OpenAI)
   482→// KEY: Uses response_format: { type: "json_object" } to force JSON output
   483→async function callOpenAICompatible(
   484→  config: ProviderConfig,
   485→  messages: SimpleAIMessage[]
   486→): Promise<SimpleAIResponse> {
   487→  const response = await fetch(`${config.baseUrl}/chat/completions`, {
   488→    method: "POST",
   489→    headers: {
   490→      "Content-Type": "application/json",
   491→      Authorization: `Bearer ${config.apiKey}`,
   492→    },
   493→    body: JSON.stringify({
   494→      model: config.model,
   495→      messages,
   496→      max_tokens: config.maxTokens,
   497→      temperature: 0.7,
   498→      response_format: { type: "json_object" },
   499→    }),
   500→  });
   501→
   502→  if (!response.ok) {
   503→    const errorText = await response.text().catch(() => "Unknown error");
   504→    throw new Error(`${config.name} API error (${response.status}): ${errorText}`);
   505→  }
   506→
   507→  const data = await response.json();
   508→  const choice = data.choices?.[0];
   509→
   510→  if (!choice?.message?.content) {
   511→    throw new Error(`${config.name} returned empty response`);
   512→  }
   513→
   514→  return {
   515→    content: choice.message.content,
   516→    provider: config.name,
   517→    model: config.model,
   518→    usage: data.usage
   519→      ? {
   520→          promptTokens: data.usage.prompt_tokens,
   521→          completionTokens: data.usage.completion_tokens,
   522→          totalTokens: data.usage.total_tokens,
   523→        }
   524→      : undefined,
   525→  };
   526→}

   528→// Gemini native API call
   529→async function callGemini(
   530→  config: ProviderConfig,
   531→  messages: SimpleAIMessage[]
   532→): Promise<SimpleAIResponse> {
   533→  const contents = messages
   534→    .filter((m) => m.role !== "system")
   535→    .map((m) => ({
   536→      role: m.role === "assistant" ? "model" : "user",
   537→      parts: [{ text: m.content }],
   538→    }));
   539→
   540→  const systemMessage = messages.find((m) => m.role === "system");
   541→
   542→  const body: Record<string, unknown> = {
   543→    contents,
   544→    generationConfig: {
   545→      maxOutputTokens: config.maxTokens,
   546→      temperature: 0.7,
   547→      responseMimeType: "application/json",
   548→    },
   549→  };
   550→
   551→  if (systemMessage) {
   552→    body.systemInstruction = {
   553→      parts: [{ text: systemMessage.content }],
   554→    };
   555→  }
   556→
   557→  const response = await fetch(
   558→    `${config.baseUrl}/models/${config.model}:generateContent?key=${config.apiKey}`,
   559→    {
   560→      method: "POST",
   561→      headers: { "Content-Type": "application/json" },
   562→      body: JSON.stringify(body),
   563→      signal: AbortSignal.timeout(GEMINI_LIMITS.REQUEST_TIMEOUT_MS),
   564→    }
   565→  );
   566→
   567→  if (!response.ok) {
   568→    const errorText = await response.text().catch(() => "Unknown error");
   569→    throw new Error(`Gemini API error (${response.status}): ${errorText}`);
   570→  }
   571→
   572→  const data = await response.json();
   573→  const content = data.candidates?.[0]?.content?.parts?.[0]?.text;
   574→
   575→  if (!content) {
   576→    const blockReason = data?.promptFeedback?.blockReason;
   577→    if (blockReason) {
   578→      throw new Error(`AI response blocked: ${blockReason}`);
   579→    }
   580→    throw new Error("Gemini returned empty response");
   581→  }
   582→
   583→  return {
   584→    content,
   585→    provider: config.name,
   586→    model: config.model,
   587→    usage: data.usageMetadata
   588→      ? {
   589→          promptTokens: data.usageMetadata.promptTokenCount,
   590→          completionTokens: data.usageMetadata.candidatesTokenCount,
   591→          totalTokens: data.usageMetadata.totalTokenCount,
   592→        }
   593→      : undefined,
   594→  };
   595→}
   596→
   597→// ═══════════════════════════════════════════
   598→// 4. MAIN: SEND MESSAGE WITH AUTO-FALLBACK
   599→// ═══════════════════════════════════════════
   600→
   601→/**
   602→ * Send a message to the AI with automatic provider fallback.
   603→ * Tries providers in order: Groq → xAI → OpenAI → Gemini
   604→ * If one fails, it tries the next.
   605→ *
   606→ * This is the main entry point — returns raw content (may be JSON).
   607→ */
   608→export async function sendMessage(
   609→  messages: SimpleAIMessage[],
   610→  user?: Partial<AppUser> | null
   611→): Promise<SimpleAIResponse> {
   612→  // Build system prompt if not already present
   613→  const hasSystemPrompt = messages.some((m) => m.role === "system");
   614→  const allMessages = hasSystemPrompt
   615→    ? messages
   616→    : [
   617→        {
   618→          role: "system" as const,
   619→          content: buildLanguageTutorSystemPrompt({
   620→            name: user?.name,
   621→            nativeLanguage: user?.nativeLanguage,
   622→            targetLanguage: user?.targetLanguage,
   623→            proficiencyLevel: user?.proficiencyLevel,
   624→          }),
   625→        },
   626→        ...messages,
   627→      ];
   628→
   629→  const configs = getProviderConfigs();
   630→
   631→  if (configs.length === 0) {
   632→    throw new Error(
   633→      "No AI providers configured. Please set at least one API key (GROQ_API_KEY, XAI_API_KEY, OPENAI_API_KEY, or GEMINI_API_KEY)."
   634→    );
   635→  }
   636→
   637→  const errors: string[] = [];
   638→
   639→  for (const config of configs) {
   640→    try {
   641→      console.log(`[AI] Trying ${config.name} (${config.model})...`);
   642→
   643→      const result =
   644→        config.name === "gemini"
   645→          ? await callGemini(config, allMessages)
   646→          : await callOpenAICompatible(config, allMessages);
   647→
   648→      console.log(`[AI] ✅ ${config.name} succeeded (${result.usage?.totalTokens ?? "?"} tokens)`);
   649→      return result;
   650→    } catch (error) {
   651→      const errMsg = error instanceof Error ? error.message : String(error);
   652→      console.warn(`[AI] ❌ ${config.name} failed: ${errMsg}`);
   653→      errors.push(`${config.name}: ${errMsg}`);
   654→    }
   655→  }
   656→
   657→  throw new Error(`All AI providers failed:\n${errors.join("\n")}`);
   658→}

   660→/**
   661→ * Check which providers are available (have API keys configured).
   662→ */
   663→export function getAvailableProviders(): string[] {
   664→  return getProviderConfigs().map((c) => c.name);
   665→}

   671→/**
   672→ * The GeminiProvider class — implements the AIProvider interface
   673→ * for backwards compatibility with existing code that uses getAIProvider().chat()
   674→ */
   675→class GeminiProvider implements AIProvider {
   676→  name = "gemini";
   677→  private apiKey: string;
   678→  private model: string;
   679→
   680→  constructor() {
   681→    this.apiKey = process.env.GEMINI_API_KEY || "";
   682→    this.model = "gemini-2.0-flash";
   683→
   684→    if (!this.apiKey) {
   685→      console.warn("[Gemini] GEMINI_API_KEY not set. AI features will not work.");
   686→    }
   687→  }
   688→
   689→  async chat(context: AIChatContext): Promise<ParsedAIResponse> {
   690→    if (!this.apiKey) {
   691→      throw new Error("GEMINI_API_KEY is not configured");
   692→    }
   693→
   694→    const contents = context.history.map((msg) => ({
   695→      role: msg.role === "assistant" ? "model" : "user",
   696→      parts: [{ text: msg.content }],
   697→    }));
   698→
   699→    contents.push({
   700→      role: "user",
   701→      parts: [{ text: context.userMessage }],
   702→    });
   703→
   704→    const url = `https://generativelanguage.googleapis.com/v1beta/models/${this.model}:generateContent?key=${this.apiKey}`;
   705→
   706→    const response = await fetch(url, {
   707→      method: "POST",
   708→      headers: { "Content-Type": "application/json" },
   709→      body: JSON.stringify({
   710→        contents,
   711→        systemInstruction: {
   712→          parts: [{ text: context.systemPrompt }],
   713→        },
   714→        generationConfig: {
   715→          temperature: 0.7,
   716→          topP: 0.95,
   717→          topK: 40,
   718→          maxOutputTokens: 2048,
   719→          responseMimeType: "application/json",
   720→        },
   721→        safetySettings: [
   722→          { category: "HARM_CATEGORY_HARASSMENT", threshold: "BLOCK_MEDIUM_AND_ABOVE" },
   723→          { category: "HARM_CATEGORY_HATE_SPEECH", threshold: "BLOCK_MEDIUM_AND_ABOVE" },
   724→          { category: "HARM_CATEGORY_SEXUALLY_EXPLICIT", threshold: "BLOCK_MEDIUM_AND_ABOVE" },
   725→          { category: "HARM_CATEGORY_DANGEROUS_CONTENT", threshold: "BLOCK_MEDIUM_AND_ABOVE" },
   726→        ],
   727→      }),
   728→      signal: AbortSignal.timeout(GEMINI_LIMITS.REQUEST_TIMEOUT_MS),
   729→    });
   730→
   731→    if (!response.ok) {
   732→      const errorText = await response.text().catch(() => "Unknown error");
   733→      throw new Error(`Gemini API error (${response.status}): ${errorText}`);
   734→    }
   735→
   736→    const data = await response.json();
   737→    const text = data?.candidates?.[0]?.content?.parts?.[0]?.text;
   738→
   739→    if (!text) {
   740→      const blockReason = data?.promptFeedback?.blockReason;
   741→      if (blockReason) {
   742→        throw new Error(`AI response blocked: ${blockReason}`);
   743→      }
   744→      throw new Error("AI returned an empty response");
   745→    }
   746→
   747→    return parseAIResponse(text);
   748→  }
   749→}

   751→// Multi-provider adapter that uses sendMessage() with fallback
   752→class FallbackProvider implements AIProvider {
   753→  name = "multi-provider";
   754→
   755→  async chat(context: AIChatContext): Promise<ParsedAIResponse> {
   756→    const messages: SimpleAIMessage[] = [
   757→      { role: "system", content: context.systemPrompt },
   758→      ...context.history.map((m) => ({
   759→        role: m.role as "user" | "assistant",
   760→        content: m.content,
   761→      })),
   762→      { role: "user", content: context.userMessage },
   763→    ];
   764→
   765→    const result = await sendMessage(messages, {
   766→      nativeLanguage: context.nativeLanguage,
   767→      targetLanguage: context.targetLanguage,
   768→      proficiencyLevel: context.proficiencyLevel,
   769→    });
   770→
   771→    // Try to parse as structured JSON, fallback to plain text
   772→    return parseAIResponse(result.content);
   773→  }
   774→}

   776→// Provider factory (backwards compatible)
   777→let providerInstance: AIProvider | null = null;
   778→
   779→/**
   780→ * Get the current AI provider instance.
   781→ * Now uses multi-provider with fallback instead of Gemini-only.
   782→ */
   783→export function getAIProvider(): AIProvider {
   784→  if (!providerInstance) {
   785→    // Use the multi-provider fallback system
   786→    providerInstance = new FallbackProvider();
   787→  }
   788→  return providerInstance;
   789→}
